# -*- coding: utf-8 -*- 

""" 

Esse arquivo contem a definição do modelo para Item na nota fiscal 

 

""" 

 

from django.db import models 

from base.models.ncm import NCM 

 

ORIGEM_CHOICES = ( 

('0', 'Nacional - Exceto código 3, 4, 5 e 8'), 

('1', 'Estrangeira - Importação direta, exceto código 6'), 

('2', 'Estrangeira - Adquirida no mercado interno, exceto código 7'), 

('3', 'Nacional - Mercadoria ou bem com conteúdo de exportação entre 40 e 70%'), 

('4', 'Nacional - Produção em conformidade com a legislação citada nos Ajustes'), 

('5', 'Nacional - Mercadoria ou bem com conteúudo de exportação menor que 40%'), 

('6', 'Estrangeira - Importação direta, sem similar nacional, constante em lista da CAMEX e gás natural'), 

('7', 'Estrangeira - Adquirida no mercado interno, sem similar nacional, constante em lista da CAMEX e gás natural'), 

('8', 'Nacional - Mercadoria ou bem') 

) 

 

TPOP_CHOICES = ( 

('0', 'Outros'), 

('1', 'Venda concessionária'), 

('2', 'Faturamento direto para consumidor final'), 

('3', 'Venda direta para grandes consumidores (frotista, governo,...)') 

) 

""" Opções de escolha para Origem do produto e Tipo de Operação de Veículos""" 

 

class Item(models.Model): 

"""Modelo para Item na nota fiscal, cada item corresponde a um produto. 

@author Bruno Normande 

""" 

 

nfe = models.ForeignKey('base.NotaFiscalEletronica', on_delete=models.PROTECT) 

n_item = models.IntegerField(verbose_name="Número do Item na nota", default=1) 

 

ncm = models.ForeignKey(NCM, on_delete=models.PROTECT) 

desc = models.CharField(verbose_name="Descrição", max_length=200) 

cfop = models.CharField(verbose_name="CFOP", max_length=5) 

cod = models.CharField(verbose_name="Código do Produto", max_length=100) 

 

cest = models.CharField(verbose_name="CEST", max_length=12, blank=True) 

gtin = models.CharField(verbose_name="GTIN/EAN", max_length=15, blank=True) 

gtin_trib = models.CharField(verbose_name="GTIN/EAN Tributado", max_length=15, blank=True) 

 

unidade_tributario = models.CharField(verbose_name="Unidade Tributado", max_length=200) 

quantidade_tributario = models.FloatField(verbose_name="Quantidade Tributado") 

valor_unitario_tributario = models.FloatField(verbose_name="Valor Unitário Tributado") 

 

unidade_comercial = models.CharField(verbose_name="Unidade Tributado", max_length=200) 

quantidade_comercial = models.FloatField(verbose_name="Quantidade Tributado") 

valor_unitario_comercial = models.FloatField(verbose_name="Valor Unitário Tributado") 

 

origem = models.CharField(verbose_name="Origem da Mercadoria", max_length=1, choices=ORIGEM_CHOICES) 

cst = models.CharField(verbose_name="Código da Situação Tributária", max_length=200, blank=True) 

situacao_simples = models.CharField(verbose_name="Situação da Operação Simples Nacional", max_length=3, blank=True) 

modalidade_bc = models.CharField(verbose_name="Modalidade da BC do ICMS", max_length=200, blank=True) 

modalidade_bc_st = models.CharField(verbose_name="Modalidade da BC do ICMS ST", max_length=200, blank=True) 

recopi = models.CharField(verbose_name="RECOPI", max_length=100, blank=True) 

cod_anp = models.CharField(verbose_name="Código de Produto da ANP", max_length=100, blank=True) 

tipo_oper = models.CharField(verbose_name="Tipo de Operação Veículos", max_length=1, 

choices=TPOP_CHOICES, blank=True) 

 

valor_icms = models.FloatField(verbose_name="Valor do ICMS", default=0.0) 

aliq_icms = models.FloatField(verbose_name="Aliquota do Imposto", default=0.0) 

valor_icms_st = models.FloatField(verbose_name="Valor do ICMS para ST", default=0.0) 

aliq_icms_st = models.FloatField(verbose_name="Aliquota do ICMS para ST", default=0.0) 

base_calculo_icms = models.FloatField(verbose_name="Base de Cálculo do ICMS", default=0.0) 

base_calculo_icms_st = models.FloatField(verbose_name="Base de Cálculo do ICMS para ST", default=0.0) 

valor_ipi = models.FloatField(verbose_name="Valor do IPI", default = 0.0) 

aliq_ipi = models.FloatField(verbose_name="Aliquota do IPI", default=0.0) 

base_calculo_ipi = models.FloatField(verbose_name="Base de calculo do IPI", default=0.0) 

 

valor_desconto = models.FloatField(verbose_name="Desconto", default=0.0) 

valor_seguro = models.FloatField(verbose_name="Seguro do item", default=0.0) 

valor_frete = models.FloatField(verbose_name="Frete do item", default=0.0) 

valor_outro = models.FloatField(verbose_name="Outros valores", default=0.0) 

 

created_at = models.DateTimeField(auto_now_add=True) 

updated_at = models.DateTimeField(auto_now=True) 

 

def __str__(self): 

return "{} - {}".format(self.ncm.codigo, self.desc) 

 

class Meta: 

app_label = 'base' 

ordering = ['-updated_at', 'ncm', ] 

verbose_name = 'Item' 

verbose_name_plural = 'Itens'

def autocomplete_search_fields(): 

return 'desc',
