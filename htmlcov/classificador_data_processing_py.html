import pandas as pd 

import numpy as np 

from sklearn.preprocessing import MinMaxScaler 

 

def normalize(data, features): 

''' Normaliza as colunas de um dataframe no intervalo 0-1  

Args: 

data: um dataframe pandas 

features: array com os nomes das colunas a serem normalizadas 

 

Returns: 

data: dataframe com os dados já normalizados 

''' 

mms = MinMaxScaler() 

data[features] = mms.fit_transform(data[features]) 

return data 

 

 

def sample(data, category, classes): 

''' Faz uma amostragem dos dados. Seleciona apenas dados das classes rele- 

vantes e uma quantidade igual da cada categoria. 

Args: 

data: um dataframe pandas 

category: coluna de categoria dos dados (no caso, a codificação) 

classes: as classes que se deseja agrupar 

 

Returns: 

sample_data: dataframe pandas correspondente à amostragem 

 

''' 

sample_data = data[data[category].isin(classes)] 

temp = sample_data.groupby(category) 

sample_data = temp.apply(lambda x: x.sample(temp.size().min()).reset_index(drop=True)) 

return sample_data 

 

def category_to_numeric(data, category, colname): 

''' Converte um atributo categórico em um numérico 

Args: 

data: um dataframe pandas 

category: coluna correspondente à categoria  

colname: nome da nova coluna 

 

Returns: 

data: dataframe com a nova coluna numérica 

''' 

data[colname] = data[category].astype('category').cat.codes 

return data

def validate_gtin(gtin):

produto_escalar = np.sum(vec_mult * vec_gtin.astype(int)) 

prox_decimal = int(produto_escalar / 10) * 10 + 10
