#!/usr/bin/env python 

# -*- coding: utf-8 -*- 

""" 

Script para importar notas do banco de dados da SEFAZ 

para o BD de nossa aplicação. 

 

@author: Bruno Normande 

""" 

 

import os, django 

import logging 

 

# Configs 

from settings import IMPORT_LOG_FILE 

 

django_mode = os.environ.get('DJANGO_MODE') 

os.environ.setdefault("DJANGO_SETTINGS_MODULE", "settings"+("" if django_mode is None else "."+django_mode)) 

django.setup() 

 

from django.db import connections, transaction 

from base.models.nota_fiscal_eletronica import NotaFiscalEletronica 

from base.models.cadastro import Cadastro 

from base.models.item import Item 

from base.models.ncm import NCM 

 

if django_mode == 'staging': 

connection = connections['prod'] 

else: 

connection = connections['default'] 

 

sql_itens_tabela = """SELECT produto.SEQ_PROD, produto.SEQ_NFE, produto.DSC_PROD, produto.COD_PROD, produto.COD_CEST, 

produto.COD_NCM, produto.COD_GTIN, produto.COD_GTIN_TRIB, produto.COD_UN_TRIB, produto.NUM_QTD_TRIB,  

produto.VAL_UN_TRIB, produto.COD_CFOP, produto.VAL_BRUTO, produto.VAL_FRETE, produto.VAL_DESC, 

produto.VAL_DESP_ACESSORIAS, produto.VAL_SEGURO, icms.COD_ORIGEM, icms.COD_CST, icms.COD_SIT_OPERACAO_SN, 

icms.COD_MOD_BC, icms.COD_MOD_BCST, icms.VAL_ALIQ_ICMS, icms.VAL_ICMS, icms.VAL_ALIQ_ICMSST, icms.VAL_ICMSST, 

icms.VAL_BC_ICMS, icms.VAL_BC_ICMSST, ipi.VAL_ALIQ_IPI, ipi.VAL_IPI, ipi.VAL_BC_IPI, comb.COD_ANP, veic.TIP_OPERACAO 

FROM 

"ADMNFE001"."NFE_PRODUTO" produto 

left join ADMNFE001.NFE_IMPOSTO_IPI ipi on ipi.SEQ_PROD = produto.SEQ_PROD 

left join ADMNFE001.NFE_IMPOSTO_ICMS icms on icms.seq_prod = produto.SEQ_PROD 

left join ADMNFE001.NFE_COMBUSTIVEL comb on comb.seq_prod = produto.SEQ_PROD 

left join ADMNFE001.NFE_VEICULO veic on veic.seq_prod = produto.SEQ_PROD 

WHERE SEQ_NFE = %s 

ORDER BY SEQ_PROD ASC 

""" 

 

sql_itens_view = """SELECT SEQ_PROD, SEQ_NFE, COD_NCM, DSC_PROD, COD_CFOP, COD_PROD, COD_CEST, COD_GTIN, 

COD_GTIN_TRIB, COD_UN_TRIB, NUM_QTD_TRIB, VAL_UN_TRIB, COD_ORIGEM, COD_CST, COD_SIT_OPERACAO_SN, 

COD_MOD_BC, COD_MOD_BCST, COD_ANP, TIP_OPERACAO, VAL_ICMS, VAL_ALIQ_ICMS, VAL_ICMSST, VAL_ALIQ_ICMSST, 

VAL_BC_ICMS, VAL_BC_ICMSST, VAL_IPI, VAL_ALIQ_IPI, VAL_BC_IPI, VAL_DESC, VAL_SEGURO, VAL_FRETE,  

VAL_DESP_ACESSORIAS, COD_UN_COM, NUM_QTD_COM, VAL_UN_TRIB 

FROM "ADMNFE001"."VW_PRODUTOS_CALC_UFAL" 

WHERE SEQ_NFE = %s  

ORDER BY SEQ_PROD ASC 

""" 

 

sql_notas_tabela = """SELECT notas.SEQ_NFE, notas.COD_CHAVE, notas.NUM_NF, notas.DAT_EMISSAO, notas.DAT_SAIDA, 

notas.TIP_NF, notas.TIP_FIN_NFE, notas.COD_STATUS, notas.COD_UF_EMISSOR, trans.COD_MOD_FRETE, notas.COD_UF_EMIT, 

notas.TIP_DOC_EMIT, notas.NUM_DOC_EMIT, notas.NUM_IE_EMIT, notas.COD_UF_DEST, notas.TIP_DOC_DEST, notas.NUM_DOC_DEST, 

notas.NUM_IE_DEST, notas.VAL_TOTAL_NF, notas.DSC_NAT_OPERACAO , emit.DSC_RAZAO_SOCIAL DSC_RAZAO_SOCIAL_EMIT, 

emit.COD_MUNICIPIO COD_MUNICIPIO_EMIT, emit.COD_REG_TRIB COD_REG_TRIB_EMIT, dest.DSC_RAZAO_SOCIAL DSC_RAZAO_SOCIAL_DEST, 

dest.COD_MUNICIPIO COD_MUNICIPIO_DEST, totais.VAL_BC_ICMS, totais.VAL_ICMS, totais.VAL_BC_ICMS_ST, totais.VAL_ICMS_ST, 

totais.VAL_TOTAL_ICMS_FCP, totais.VAL_TOTAL_ICMS_UF_DESTINO, totais.VAL_TOTAL_ICMS_UF_REMETENTE 

FROM 

ADMNFE001.NFE_NOTAS notas 

inner join ADMNFE001.NFE_TOTAIS totais on notas.SEQ_NFE = totais.SEQ_NFE 

inner join ADMNFE001.NFE_EMITENTE emit on notas.SEQ_NFE = emit.SEQ_NFE 

inner join ADMNFE001.NFE_DESTINATARIO dest on notas.SEQ_NFE = dest.SEQ_NFE 

left join ADMNFE001.NFE_TRANSPORTE trans on trans.SEQ_NFE = notas.SEQ_NFE 

left join ADMLOG001.LOG_MUNICIPIO mun on emit.COD_MUNICIPIO = mun.COD_IBGE; 

WHERE 

DTH_INCLUSAO >= TRUNC(SYSDATE) - INTERVAL '1' DAY AND 

DTH_INCLUSAO < TRUNC(SYSDATE) AND 

((TIP_NF='0' AND COD_UF_EMIT='AL' AND COD_UF_DEST<>'AL') OR (TIP_NF='1' AND COD_UF_EMIT<>'AL' AND COD_UF_DEST='AL')) 

ORDER BY DAT_EMISSAO ASC 

""" 

 

sql_notas_view = """SELECT SEQ_NFE, COD_CHAVE, NUM_NF, DAT_EMISSAO, DAT_SAIDA, TIP_NF, TIP_FIN_NFE, COD_STATUS, 

COD_UF_EMISSOR, COD_MOD_FRETE, COD_UF_EMIT, TIP_DOC_EMIT, NUM_DOC_EMIT, NUM_IE_EMIT, COD_UF_DEST, TIP_DOC_DEST, 

NUM_DOC_DEST, NUM_IE_DEST, VAL_TOTAL_NF, DSC_RAZAO_SOCIAL_EMIT, COD_MUNICIPIO_EMIT, COD_REG_TRIB_EMIT, 

DSC_RAZAO_SOCIAL_DEST, COD_MUNICIPIO_DEST, DSC_NAT_OPERACAO, DTH_INCLUSAO, DTH_ALTERACAO, VAL_TOTAL_ICMS_UF_DESTINO, 

VAL_TOTAL_ICMS_FCP 

FROM "ADMNFE001"."VW_NOTAS_FISCAIS_CALC_UFAL" 

WHERE 

DTH_INCLUSAO > SYSDATE - INTERVAL '9' HOUR AND 

DTH_ALTERACAO < SYSDATE - INTERVAL '1' HOUR AND 

((TIP_NF='0' AND COD_UF_EMIT='AL' AND COD_UF_DEST<>'AL') OR (TIP_NF='1' AND COD_UF_EMIT<>'AL' AND COD_UF_DEST='AL')) 

ORDER BY SEQ_NFE ASC 

""" 

 

def main(): 

logging.basicConfig( 

format="%(asctime)s - %(levelname)s - %(message)s", 

level=logging.INFO, 

filename=IMPORT_LOG_FILE 

) 

 

last_nfe = get_last_nfe_id() 

logging.info("Iniciando importação do banco SEFAZ do SEQ_NFE {}".format(last_nfe)) 

 

with connection.cursor() as cursor: 

cursor.execute(sql_notas_view) 

all_nfe = cursor.fetchall() 

logging.info("Importando {} notas".format(len(all_nfe))) 

for a_nfe in all_nfe: 

with transaction.atomic(): 

nfe_seq = a_nfe[0] 

 

if NotaFiscalEletronica.objects.filter(chave_nota=a_nfe[1]).exists(): 

logging.info("Nota já existe!") 

continue; 

else: 

logging.info("Importando nota com chave {} - seq {} - inc {} - alt {}".format(a_nfe[1],a_nfe[0],a_nfe[25],a_nfe[26])) 

 

emit, emit_created = Cadastro.objects.get_or_create( 

doc=a_nfe[12], 

uf=a_nfe[10], 

crt=a_nfe[21], 

defaults={ 

"cod_municipio": a_nfe[20], 

"razao": a_nfe[19], 

"ie": a_nfe[13] 

} 

) 

dest, dest_created = Cadastro.objects.get_or_create( 

doc=a_nfe[16], 

uf=a_nfe[14], 

crt='', 

defaults={ 

"cod_municipio": a_nfe[23], 

"razao": a_nfe[22], 

"ie": a_nfe[17] if a_nfe[17] else '', 

} 

) 

 

try: 

nfe = NotaFiscalEletronica.objects.create( 

chave_nota=a_nfe[1], 

num_nota=a_nfe[2], 

data_emissao=a_nfe[3], 

data_saida=a_nfe[4], 

tipo=a_nfe[5], 

nat_oper=a_nfe[24], 

finalidade=a_nfe[6], 

emitente=emit, 

destinatario=dest, 

uf_cod=a_nfe[8], 

mod_frete=a_nfe[9] if a_nfe[9] else '9', 

valor_total=a_nfe[18], 

val_total_icms_uf_destino=a_nfe[27] if a_nfe[27] else 0.0, 

val_total_icms_fcp=a_nfe[28] if a_nfe[28] else 0.0 

) 

except: 

logging.exception("Erro ao importar nota") 

continue 

 

cursor.execute(sql_itens_view, [nfe_seq]) 

n_item = 1 

for i in cursor.fetchall(): 

logging.debug("-- Importando item {}, NCM: {}".format(i[0], i[2])) 

ncm, ncm_created = NCM.objects.get_or_create( 

codigo=i[2] 

) 

Item.objects.create( 

nfe=nfe, 

n_item=n_item, 

ncm=ncm, 

desc=i[3] if i[3] else 'X', 

cfop=i[4], 

cod=i[5], 

cest=i[6] if len(str(i[6])) == 7 else '0'+str(i[6]) if len(str(i[6])) == 6 else '', 

gtin=i[7] if i[7] else '', 

gtin_trib=i[8] if i[8] else i[7], 

 

origem=i[12] if i[12] else 0, 

cst=i[13], 

situacao_simples=i[14], 

modalidade_bc=i[15] if i[15] else '', 

modalidade_bc_st=i[16] if i[16] else '', 

cod_anp=i[17], 

tipo_oper=i[18], 

 

unidade_tributario=i[9], 

quantidade_tributario=i[10], 

valor_unitario_tributario=i[11], 

unidade_comercial=i[32], 

quantidade_comercial=i[33], 

valor_unitario_comercial=i[34], 

 

valor_icms=i[19] if i[19] else 0, 

aliq_icms=i[20] if i[20] else 0, 

valor_icms_st=i[21] if i[21] else 0, 

aliq_icms_st=i[22] if i[22] else 0, 

base_calculo_icms=i[23] if i[23] else 0, 

base_calculo_icms_st=i[24] if i[24] else 0, 

valor_ipi=i[25] if i[25] else 0, 

aliq_ipi=i[26] if i[26] else 0, 

base_calculo_ipi=i[27] if i[27] else 0, 

 

valor_desconto=i[28] if i[28] else 0, 

valor_seguro=i[29] if i[29] else 0, 

valor_frete=i[30] if i[30] else 0, 

valor_outro=i[31] if i[31] else 0 

) 

n_item += 1 

 

logging.info("Encerrando importador") 

 

 

def get_last_nfe_id(): 

try: 

ultima_importada = NotaFiscalEletronica.objects.latest(field_name="created_at") 

except NotaFiscalEletronica.DoesNotExist: 

return 0 

else: 

ultima_importada = ultima_importada.chave_nota 

cur = connection.cursor() 

cur.execute("""SELECT SEQ_NFE FROM "ADMNFE001"."VW_NOTAS_FISCAIS_CALC_UFAL"  

WHERE COD_CHAVE = %s """, [ultima_importada]) 

result = cur.fetchone() 

return result[0] 

 

if __name__ == '__main__': 

main()
