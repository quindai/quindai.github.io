class CACEAL(models.Model): 

"""Modelo para CACEAL. 

@author Bruno Normande 

""" 

 

numero = models.CharField(verbose_name="Número do CACEAL", max_length=8, unique=True) 

cnpj = models.CharField(verbose_name="CNPJ", max_length=14) 

razao = models.CharField(verbose_name="Razão Social", max_length=200, blank=True) 

ind_simples = models.CharField(verbose_name="Indica se é contribuinte do Simples", max_length=1) 

cod_regime = models.CharField(verbose_name="Código do regime de pagamento", choices=REGIME_CHOICES, max_length=3) 

cod_situacao = models.CharField(verbose_name="Código da situação cadastral", choices=SITUACAO_CHOICES, max_length=2) 

tipo_contrib = models.CharField(verbose_name="Código do tipo de contribuinte", max_length=2) 

data_inicio = models.DateField(verbose_name="Data de início da atividade") 

 

def __str__(self): 

return "{}: {}-{}".format(self.cnpj, self.numero, self.cod_situacao)
